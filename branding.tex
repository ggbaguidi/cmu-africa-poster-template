% Individual logo helper. Each supplied file is included proportionally and is
% never cropped, recolored, or reconstructed.
\newcommand{\InstitutionLogo}[2][55mm]{%
  \IfFileExists{#2}{\includegraphics[width=#1,height=25mm,keepaspectratio]{#2}}{%
    \fbox{\parbox[c][20mm][c]{#1}{\centering\PosterSmallFont Missing logo\\\texttt{#2}}}%
  }%
}

% ---------- Automatically managed institutional-logo row ----------
% Preferred usage in main.tex:
%   \AddInstitutionLogo{logos/institution-a.pdf}
%   \AddInstitutionLogo{logos/institution-b.pdf}
% The number of logos is detected automatically. They are evenly distributed in
% a reserved header row, so the title and columns never move. The optional width
% overrides automatic sizing for an unusually narrow or wide mark:
%   \AddInstitutionLogo[52mm]{logos/sponsor.pdf}
\ExplSyntaxOn
\seq_new:N \g_cmu_institution_logos_seq
\dim_new:N \g_cmu_logo_row_width_dim
\dim_new:N \g_cmu_logo_gap_dim
\dim_new:N \g_cmu_logo_max_width_dim
\dim_new:N \g_cmu_logo_max_height_dim
\dim_new:N \l_cmu_logo_auto_width_dim
\dim_gset:Nn \g_cmu_logo_row_width_dim { 520mm }
\dim_gset:Nn \g_cmu_logo_gap_dim { 12mm }
\dim_gset:Nn \g_cmu_logo_max_width_dim { 72mm }
\dim_gset:Nn \g_cmu_logo_max_height_dim { 23mm }

\NewDocumentCommand{\AddInstitutionLogo}{O{}m}
  { \seq_gput_right:Nn \g_cmu_institution_logos_seq { \cmu_logo_item:nn {#1}{#2} } }
\NewDocumentCommand{\ClearInstitutionLogos}{}
  { \seq_gclear:N \g_cmu_institution_logos_seq }
\NewDocumentCommand{\SetInstitutionLogoRowWidth}{m}
  { \dim_gset:Nn \g_cmu_logo_row_width_dim {#1} }
\NewDocumentCommand{\SetInstitutionLogoGap}{m}
  { \dim_gset:Nn \g_cmu_logo_gap_dim {#1} }
\NewDocumentCommand{\SetInstitutionLogoMaxSize}{mm}
  {
    \dim_gset:Nn \g_cmu_logo_max_width_dim {#1}
    \dim_gset:Nn \g_cmu_logo_max_height_dim {#2}
  }
\NewExpandableDocumentCommand{\InstitutionLogoCount}{}
  { \int_to_arabic:n { \seq_count:N \g_cmu_institution_logos_seq } }

\cs_new_protected:Npn \cmu_logo_item:nn #1#2
  {
    \IfFileExists{#2}
      {
        \tl_if_blank:nTF {#1}
          { \includegraphics[width=\l_cmu_logo_auto_width_dim,
              height=\g_cmu_logo_max_height_dim,keepaspectratio]{#2} }
          { \includegraphics[width=#1,height=\g_cmu_logo_max_height_dim,
              keepaspectratio]{#2} }
      }
      {
        \fbox{\parbox[c][18mm][c]{\l_cmu_logo_auto_width_dim}
          {\centering\fontsize{14}{17}\selectfont Missing logo\\\texttt{#2}}}
      }
  }

\cs_new_protected:Npn \cmu_print_institution_logos:
  {
    \int_compare:nNnT { \seq_count:N \g_cmu_institution_logos_seq } > { 0 }
      {
        \dim_set_eq:NN \l_cmu_logo_auto_width_dim
          \g_cmu_logo_row_width_dim
        \int_step_inline:nn
          { \seq_count:N \g_cmu_institution_logos_seq - 1 }
          { \dim_sub:Nn \l_cmu_logo_auto_width_dim { \g_cmu_logo_gap_dim } }
        \dim_set:Nn \l_cmu_logo_auto_width_dim
          {
            \l_cmu_logo_auto_width_dim
            / \int_eval:n { \seq_count:N \g_cmu_institution_logos_seq }
          }
        \dim_compare:nNnT { \l_cmu_logo_auto_width_dim } >
          { \g_cmu_logo_max_width_dim }
          { \dim_set_eq:NN \l_cmu_logo_auto_width_dim \g_cmu_logo_max_width_dim }
        \hbox_to_wd:nn { \g_cmu_logo_row_width_dim }
          {
            \hfil
            \seq_use:Nn \g_cmu_institution_logos_seq
              { \hspace{\g_cmu_logo_gap_dim} }
            \hfil
          }
      }
  }
\NewDocumentCommand{\PrintInstitutionLogos}{}
  { \cmu_print_institution_logos: }
\ExplSyntaxOff

% These PNGs were extracted from the .pptx template provided during the English

% Draw a thin tiled tartan stripe without stretching the pattern.
\newcommand{\MakeBrandFrame}{%
  \begin{tikzpicture}[remember picture,overlay]
    % CMU-Africa wordmark, top left.
    \node[anchor=north west,inner sep=0] at
      ([xshift=10mm,yshift=-7mm]current page.north west)
      {\includegraphics[width=205mm,keepaspectratio]{\CMUAfricaWordmarkFile}};

    % Top stripe begins after the wordmark, matching the provided layout.
    \begin{scope}
      \clip ([xshift=230mm,yshift=-8mm]current page.north west)
        rectangle ([xshift=-10mm,yshift=-15mm]current page.north east);
      \foreach \x in {230,370,510,650,790,930,1070}{
        \node[anchor=north west,inner sep=0] at
          ([xshift=\x mm,yshift=-8mm]current page.north west)
          {\includegraphics[width=140mm,height=6.6mm]{\CMUTartanStripeFile}};
      }
    \end{scope}

    % Bottom stripe spans the complete page width.
    \begin{scope}
      \clip (current page.south west)
        rectangle ([yshift=6.6mm]current page.south east);
      \foreach \x in {0,140,280,420,560,700,840,980,1120}{
        \node[anchor=south west,inner sep=0] at
          ([xshift=\x mm]current page.south west)
          {\includegraphics[width=140mm,height=6.6mm]{\CMUTartanStripeFile}};
      }
    \end{scope}

    % Dedicated institutional-logo row. It remains empty when no logos are added.
    \node[anchor=north,inner sep=0] at
      ([yshift=-104mm]current page.north)
      {\PrintInstitutionLogos};
  \end{tikzpicture}%
}

  \setlength{\SimpleBodyHeight}{650mm}

  \setlength{\SimpleBodyHeight}{990mm}

\newcommand{\SimplePosterColumns}[3]{%
  \noindent
  \begin{minipage}[t][\SimpleBodyHeight][s]{\dimexpr(\textwidth-2\SimpleColumnGap-2\SimpleRuleWidth)/3-.6mm\relax}
    \vspace{0pt}#1
  \end{minipage}%
  \hspace{\dimexpr\SimpleColumnGap/2\relax}%
  \begin{minipage}[t][\SimpleBodyHeight][t]{\SimpleRuleWidth}
    \vspace{0pt}\rule{\SimpleRuleWidth}{\SimpleBodyHeight}
  \end{minipage}%
  \hspace{\dimexpr\SimpleColumnGap/2\relax}%
  \begin{minipage}[t][\SimpleBodyHeight][s]{\dimexpr(\textwidth-2\SimpleColumnGap-2\SimpleRuleWidth)/3-.6mm\relax}
    \vspace{0pt}#2
  \end{minipage}%
  \hspace{\dimexpr\SimpleColumnGap/2\relax}%
  \begin{minipage}[t][\SimpleBodyHeight][t]{\SimpleRuleWidth}
    \vspace{0pt}\rule{\SimpleRuleWidth}{\SimpleBodyHeight}
  \end{minipage}%
  \hspace{\dimexpr\SimpleColumnGap/2\relax}%
  \begin{minipage}[t][\SimpleBodyHeight][s]{\dimexpr(\textwidth-2\SimpleColumnGap-2\SimpleRuleWidth)/3-.6mm\relax}
    \vspace{0pt}#3
  \end{minipage}%
}
